async function checkProgress() { const login = document.getElementById('loginNumber').value.trim().toUpperCase(); const resultDiv = document.getElementById('progress-result'); resultDiv.innerHTML = 'Loading...'; if (!login) { resultDiv.innerHTML = '

Please enter a client number.

'; return; } try { const res = await fetch('https://goglobal-recruitment.github.io/Go-Global-Recruitment/fixed_data.csv'); const txt = await res.text(); const data = parseCSV(txt); const client = data.find(c => c['Client Number']?.trim().toUpperCase() === login); if (!client) { resultDiv.innerHTML = '

Client number not found. Please check and try again.

'; return; } resultDiv.innerHTML = `

Status for ${login}

Email: interviewsglo@outlook.com

  • CV Uploaded:${formatStatus(client['CV Uploaded'])}
  • Offer Letter:${formatStatus(client['Offer Letter'])}
  • Service Agreement:${formatStatus(client['Go-Global Service Agreement'])}
  • Employment Contract:${formatStatus(client['Employment Contract'])}
  • Debtform/Schedule:${formatStatus(client['Debtform/Schedule'])}
  • Client Service Booking:${formatStatus(client['Client Service booking'])}
  • Biometrics Booking:${formatStatus(client['Biometrics Booking'])}
  • Medical Booked:${formatStatus(client['Medical Booked'])}
  • Psychologist Booking:${formatStatus(client['Psychologist Booking'])}
  • Elites Test Booking:${formatStatus(client['Elites Test Booking'])}
  • Hiring Meeting:${formatStatus(client['Hiring Company Meeting'])}
  • Final Status:${formatStatus(client['Final Status'])}
`; } catch (err) { console.error(err); resultDiv.innerHTML = '

Failed to load progress. Try again later.

'; } }
